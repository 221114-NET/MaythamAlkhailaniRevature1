
//We need to creat dictionary that hold frequencies and then Iterate over it.
//Every time, we trake the smallest bird type with highest frequency.
//If we find one we return it.




int migratoryBirds(int[] arr)
{
    // Creating Dictionary to hold the types. 

    Dictionary<int, int> frequencies = new Dictionary<int, int>();

    // Iterate over the array and count the frequency.


    foreach (int type in arr)
    {
        if (!frequencies.ContainsKey(type))
        {
            frequencies[type] = 1;
        }
        else
        {
            frequencies[type]++;
        }
    }

    // Find the maximum frequency


    int maxFrequency = 0;
    foreach (int freq in frequencies.Values)
    {
        maxFrequency = Math.Max(maxFrequency, freq);
    }

    // Find the smallest bird type with the maximum frequency.

    
    int minType = int.MaxValue;
    foreach (int type in frequencies.Keys)
    {
        if (frequencies[type] == maxFrequency)
        {
            minType = Math.Min(minType, type);
        }
    }

    // Return the smallest bird type with the maximum frequency.
    return minType;
}
